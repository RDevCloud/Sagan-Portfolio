/**
 * Deferring heavy API loads to prevent TBT (Total Blocking Time)
 * Ensuring Tag Manager events wait for idle state

export const optimizeLoadQueue = () => {
  // Use RequestIdleCallback for non-critical analytics
  if ('requestIdleCallback' in window) {
    requestIdleCallback(() => {
      injectGTM();
      preloadHeavyAssets();
    });
  } else {
    setTimeout(() => { injectGTM(); }, 2000);
  }
};

  const dataLayer = window.dataLayer || [];
  dataLayer.push({
